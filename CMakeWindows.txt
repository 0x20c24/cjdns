# the name of the target operating system
SET(CMAKE_SYSTEM_NAME Windows)

# which compilers to use for C and C++
SET(CMAKE_C_COMPILER i586-mingw32msvc-gcc)
SET(CMAKE_CXX_COMPILER i586-mingw32msvc-g++)
SET(CMAKE_RC_COMPILER i586-mingw32msvc-windres)

# here is the target environment located
SET(CMAKE_FIND_ROOT_PATH /usr/i586-mingw32msvc)

# --prefix is wrong in this case but I'm not sure
# how else to tell autotools to look for headers somewhere.
set(EVENT2_CONFIG --prefix=/usr/i686-w64-mingw32 --host=i586-mingw32msvc)

# windows provides clock_gettime() without librt.
set(NO_LIBRT TRUE)

# BufferAllocator.c:1:0: error: -fPIC ignored for target (all code is position independent)
set(NO_PIE TRUE)

# proceed with ifdef hell :(
add_definitions(-DWIN32)
